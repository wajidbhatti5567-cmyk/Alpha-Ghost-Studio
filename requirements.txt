import os
import logging
from email_agent import EmailAgent
from animation_engine import AnimationEngine
from thumbnail_engine import ThumbnailEngine

# Configure logging for tracking the studio process
logging.basicConfig(level=logging.INFO, filename='studio_log.log', 
                    format='%(asctime)s - %(levelname)s - %(message)s')

def run_studio():
    try:
        # Initialize the email agent to check for commands
        agent = EmailAgent()
        logging.info("Checking for new commands in email...")
        email_content = agent.check_email()
        
        if email_content:
            logging.info("Command received. Starting production...")
            
            # Step 1: Generate Script
            animator = AnimationEngine()
            script = animator.generate_animation_script(email_content)
            
            # Step 2: Process Animation
            video_path = animator.process_animation(script)
            
            # Step 3: Create Thumbnail
            thumb = ThumbnailEngine()
            thumb_path = thumb.create_thumbnail(email_content)
            
            # Step 4: Final Confirmation
            agent.send_reply("Studio Production Completed Successfully!", [video_path, thumb_path])
            logging.info("Production finished and email sent.")
        else:
            logging.info("No new commands found.")
            
    except Exception as e:
        logging.error(f"Critical error in main studio flow: {e}")

if __name__ == "__main__":
    run_studio()
